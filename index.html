<script>

  var map = L.map('map', { zoomControl: true });


  L.tileLayer('https://{s}.basemaps.cartocdn.com/light_nolabels/{z}/{x}/{y}{r}.png', {

    attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors &copy; <a href="https://carto.com/attributions">CARTO</a>',

    subdomains: 'abcd',

    maxZoom: 20

  }).addTo(map);


  map.setView([-25.2969, -57.6280], 17);


  var allLayers = [];

  var layersToLoad = 2;


  function fitAll() {

    var group = new L.featureGroup(allLayers);

    if (group.getLayers().length > 0) {

      map.fitBounds(group.getBounds(), { padding: [20, 20] });

    }

  }


  function layerDone() {

    layersToLoad--;

    if (layersToLoad <= 0) {

      document.getElementById('loading').style.display = 'none';

    }

    fitAll();

  }


  // ---- Icono rombo rojo (individuos_cha) ----

  var diamondIcon = L.divIcon({

    className: '',

    html: '<div style="width:16px;height:16px;background:#e30613;border:1px solid #7a0000;transform:rotate(45deg);"></div>',

    iconSize: [16, 16],

    iconAnchor: [8, 8]

  });


  // ---- Función genérica: abre un .gpkg y devuelve un GeoJSON FeatureCollection ----

  function gpkgToGeoJSON(url) {

    return fetch(url)

      .then(res => res.arrayBuffer())

      .then(buffer => GeoPackageAPI.open(new Uint8Array(buffer)))

      .then(geoPackage => {

        var tableName = geoPackage.getFeatureTables()[0]; // usa la primera tabla de features

        var features = [];


        geoPackage.iterateGeoJSONFeatures(tableName, feature => {

          features.push(feature);

        });


        return { type: "FeatureCollection", features: features };

      });

  }


  // ---- Carga censo_arboreo_cha (puntos verdes) ----

  gpkgToGeoJSON('censo_arboreo_cha.gpkg')

    .then(geojson => {

      var layer = L.geoJSON(geojson, {

        pointToLayer: function (feature, latlng) {

          return L.circleMarker(latlng, {

            radius: 6,

            fillColor: "#1a5d1a",

            color: "#0d3d0d",

            weight: 1,

            opacity: 1,

            fillOpacity: 0.85

          });

        }

      }).addTo(map);

      allLayers.push(layer);

      layerDone();

    })

    .catch(err => {

      console.error('Error cargando censo_arboreo_cha.gpkg:', err);

      layerDone();

    });


  // ---- Carga individuos_cha (rombos rojos + etiqueta Nro. Indiv) ----

  gpkgToGeoJSON('individuos_cha.gpkg')

    .then(geojson => {

      var layer = L.geoJSON(geojson, {

        pointToLayer: function (feature, latlng) {

          return L.marker(latlng, { icon: diamondIcon });

        },

        onEachFeature: function (feature, lyr) {

          var label = feature.properties["Nro. Indiv"] || '';

          if (label !== '') {

            lyr.bindTooltip(String(label), {

              permanent: true,

              direction: 'right',

              offset: [10, 0],

              className: 'tree-label'

            });

          }

        }

      }).addTo(map);

      allLayers.push(layer);

      layerDone();

    })

    .catch(err => {

      console.error('Error cargando individuos_cha.gpkg:', err);

      layerDone();

    });

</script>
